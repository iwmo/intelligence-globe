    Add Effect 3: AOI right-click handler (runs once when viewer is ready):
    ```typescript
    useEffect(() => {
      if (!viewer || viewer.isDestroyed()) return;
      // Create AOI point collection
      const aoiColl = viewer.scene.primitives.add(new PointPrimitiveCollection());
      aoiCollectionRef.current = aoiColl;
      const handler = new ScreenSpaceEventHandler(viewer.scene.canvas);
      (handlerRef as React.MutableRefObject<any>).current = handler;
      handler.setInputAction((click: { position: Cartesian2 }) => {
        const earthPos = viewer.scene.pickPosition(click.position);
        if (!earthPos) return;
        const carto = Ellipsoid.WGS84.cartesianToCartographic(earthPos);
        const lat = CesiumMath.toDegrees(carto.latitude);
        const lon = CesiumMath.toDegrees(carto.longitude);
        useAppStore.getState().setAreaOfInterest({ lat, lon });
        // Update AOI marker
        aoiColl.removeAll();
        aoiColl.add({ position: Cartesian3.fromDegrees(lon, lat, 0), pixelSize: 8,
          color: Color.fromCssColorString('#ffffff').withAlpha(0.9) });
      }, ScreenSpaceEventType.RIGHT_CLICK);
      return () => {
        handler.destroy();
        if (!aoiColl.isDestroyed()) viewer.scene.primitives.remove(aoiColl);
      };
    }, [viewer]);
    ```

    Add Effect 4: COMPUTE_OVERPASS dispatch (debounced 1s on replayTs change):
    ```typescript
    useEffect(() => {
      const { replayMode, replayTs, areaOfInterest, tleLastUpdated } = useAppStore.getState();
      if (replayMode !== 'playback' || !areaOfInterest || !workerRef.current) return;
      const TLE_MAX_AGE_MS = 7 * 24 * 60 * 60 * 1000;
      const tleAge = tleLastUpdated ? Date.now() - new Date(tleLastUpdated).getTime() : Infinity;
      if (tleAge > TLE_MAX_AGE_MS) return;  // suppress; TLE stale warning rendered in PlaybackBar
      const timer = setTimeout(() => {
        workerRef.current?.postMessage({
          type: 'COMPUTE_OVERPASS',
          payload: { lat: areaOfInterest.lat, lon: areaOfInterest.lon,
            timestamp: replayTs, elevationThresholdDeg: 10 },
        });
      }, 1000);
      return () => clearTimeout(timer);
    // replayTs in deps triggers effect on every scrub; 1s debounce prevents worker saturation
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [useAppStore(s => s.replayTs), useAppStore(s => s.replayMode),
        useAppStore(s => s.areaOfInterest)]);
    ```
